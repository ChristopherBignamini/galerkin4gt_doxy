Matrix Vector Multiplication

This example is focused on the calculation of a matrix-vector product

A*x=b

and can be seen as the continuation of the previous one, where we limited ourselves to an elementwise vector-vector multiplication. The main difference with respect to the previous cases is represented by storage cardinality, since in this case the matrix will require 5 indices: three indices to identify an element in the mesh (the i, j, k space coordinates) like in the vector case and two indeces for the dof pair of the considered element. In this example we will consider a matrix without any specific structure however, in real Galerkin problems, we will deal with sparse matrices: they are reinterpreted as multidimensional array stencils, with one dimension corresponding to the band. According to this interpretation, the following matrix (in which each mesh element is represented by a diagonal block) could be represented by the array 3x3x3

[11,12,13, 21,22,23, 31,32,33,
44,45,46, 54,55,56, 64,65,66,
77,78,79, 87,88,89 ,97,98,99]

The details already provided concerning vector strorage structures are of course valid also in this case. It follows that also√ü matrix-vector multiplications can be seen as stencil operations.

#include <cassert>
#include "../functors/matvec.hpp"
#include "../galerkin_defs.hpp"
#include "../numerics/basis_functions.hpp"
#include "stencil-composition/stencil-composition.hpp"
#include "stencil-composition/axis.hpp"
typedef gridtools::interval<gridtools::level<0,-2>, gridtools::level<1,1> > axis;
int main() {
constexpr unsigned int d1=5;
constexpr unsigned int d2=5;
constexpr unsigned int d3=5;
constexpr unsigned int n_rows(10);
constexpr unsigned int n_cols(20);
// Matrix A
using matrix_A_storage_info_t=gdl::storage_info< __COUNTER__, gdl::layout_tt<5> >;
using matrix_A_type=gdl::storage_t< matrix_A_storage_info_t >;
matrix_A_storage_info_t m_A_(d1,d2,d3,n_rows,n_cols);
matrix_A_type m_A(m_A_, 0.e0, "m_A");
// vector b
using vector_b_storage_info_t=gdl::storage_info< __COUNTER__, gdl::layout_tt<4> >;
using vector_b_type=gdl::storage_t< vector_b_storage_info_t >;
vector_b_storage_info_t v_b_(d1,d2,d3,n_cols);
vector_b_type v_b(v_b_, 0.e0, "v_b");
// vector out
using vector_out_storage_info_t=gdl::storage_info< __COUNTER__, gdl::layout_tt<4> >;
using vector_out_type=gdl::storage_t< vector_out_storage_info_t >;
vector_out_storage_info_t v_out_(d1,d2,d3,n_rows);
vector_out_type v_out(v_out_, 0.e0, "v_out");
for (gdl::uint_t i=0; i<d1; i++)
for (gdl::uint_t j=0; j<d2; j++)
for (gdl::uint_t k=0; k<d3; k++)
for (gdl::uint_t row=0; row<n_rows; ++row) {
for (gdl::uint_t col=0; col<n_cols; ++col)
m_A(i,j,k,row,col) = 5.0;//*(row-col);
v_b(i,j,k,row) = -1.0;
v_out(i,j,k,row) = 0.0;
}
auto grid=gridtools::grid<axis>({0, 0, 0, d1-1, d1},
{0, 0, 0, d2-1, d2});
grid.value_list[0] = 0;
grid.value_list[1] = d3-1;
typedef gdl::gt::arg<0,matrix_A_type> p_m_A;
typedef gdl::gt::arg<1,vector_b_type> p_v_b;
typedef gdl::gt::arg<2,vector_out_type> p_v_out;
typedef boost::mpl::vector<p_m_A, p_v_b, p_v_out> accessor_list;
::gridtools::aggregator_type<accessor_list> domain(boost::fusion::make_vector(&m_A, &v_b, &v_out));
auto compute=::gridtools::make_computation<BACKEND>(domain,
grid,
gridtools::make_multistage(gridtools::enumtype::execute<gridtools::enumtype::forward>(),
gridtools::make_stage<gdl::functors::matvec>(p_m_A(),p_v_b(),p_v_out())
));
compute->ready();
compute->steady();
compute->run();
compute->finalize();
bool success = true;
for (gdl::uint_t i=0; i<d1; i++)
for (gdl::uint_t j=0; j<d2; j++)
for (gdl::uint_t k=0; k<d3; k++){
for (gdl::uint_t row=0; row<n_rows; ++row){
gdl:float_t sum = 0.;
for (gdl::uint_t col=0; col<n_cols; ++col)
sum += m_A(i,j,k,row,col)*v_b(i,j,k,col);
if(sum != v_out(i,j,k,row)){
std::cout<<"["<<i<<","<<j<<","<<k<<","<<row<<"] error: "<<sum<<" != "<<v_out(i,j,k,row)<<std::endl;
success = false;
break;
}
}
}
assert(success);
}
